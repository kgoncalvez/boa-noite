import requests from bs4 import BeautifulSoup import os from urllib.parse import urljoin, urlparse def baixar_pagina(url, diretorio): resposta = requests.get(url) resposta.raise_for_status() soup = BeautifulSoup(resposta.text, 'html.parser') if not os.path.exists(diretorio): os.makedirs(diretorio) # Salvar o HTML html_path = os.path.join(diretorio, 'index.html') with open(html_path, 'w', encoding='utf-8') as f: f.write(soup.prettify()) # Baixar e salvar CSS for link in soup.find_all('link', href=True): href = link['href'] if href.endswith('.css'): css_url = urljoin(url, href) css_resposta = requests.get(css_url) css_resposta.raise_for_status() css_path = os.path.join(diretorio, os.path.basename(href)) with open(css_path, 'w', encoding='utf-8') as f: f.write(css_resposta.text) link['href'] = os.path.basename(href) # Atualizar o HTML com os caminhos relativos para CSS with open(html_path, 'w', encoding='utf-8') as f: f.write(soup.prettify()) print(f"Páginas espelhadas salvas em {diretorio}") # URL da página que você quer espelhar url = 'https://www.google.com' diretorio = 'pagina_google_espelhada' baixar_pagina(url, diretorio)
